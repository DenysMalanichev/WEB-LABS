function generateArray(){ var arr = []; for (let i = 0; i < document.getElementById("ArrayLen").value; i++) { arr.push(getRandomInt(100)); } document.getElementById("row_Array").textContent = "Не відсортований масив: " + arr; var min = Math.min.apply(null, arr); var minIndex = arr.indexOf(min); document.getElementById("min_in_Array").textContent = "Найменший елемет в масиві: " + min; arr.splice(minIndex, 1); arr.unshift(min); document.getElementById("min_start_Array").textContent = "Наймеший елемент на початку: " + arr; arr = sort(arr); document.getElementById("sorted_Array").textContent = "Відсортоваий масив: " + arr; } function getRandomInt(max) { return Math.floor(Math.random() * max); } function sort(list) { for (let i = 0; i < list.length - 1; i++) { let min = i; for (let j = i + 1; j < list.length; j++) { if (list[j] < list[min]) { min = j; } } [list[i], list[min]] = [list[min], list[i]]; } return list; }
